#!/usr/bin/env python3
import os
import json
import time
import logging
import paho.mqtt.client as mqtt
from influxdb_client import InfluxDBClient, Point
from influxdb_client.client.write_api import SYNCHRONOUS, WriteOptions

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)
logger = logging.getLogger(__name__)

# ===== 配置区域 =====
INFLUXDB_URL = os.environ.get("INFLUXDB_URL", "http://localhost:8086")
INFLUXDB_TOKEN = os.environ.get("INFLUXDB_TOKEN", "")
INFLUXDB_ORG = os.environ.get("INFLUXDB_ORG", "your_org")
INFLUXDB_BUCKET = os.environ.get("INFLUXDB_BUCKET", "your_bucket")
MQTT_BROKER = os.environ.get("MQTT_BROKER", "localhost")
MQTT_PORT = int(os.environ.get("MQTT_PORT", "1883"))
MQTT_USER = os.environ.get("MQTT_USER", "")
MQTT_PASS = os.environ.get("MQTT_PASS", "")
MQTT_TOPICS = ["device/response", "device/telemetry", "device/alert", "device/heartbeat"]

if not INFLUXDB_TOKEN:
    logger.error("INFLUXDB_TOKEN 环境变量未设置")
    exit(1)
if not MQTT_PASS:
    logger.error("MQTT_PASS 环境变量未设置")
    exit(1)

# ========================
influx_client = InfluxDBClient(
    url=INFLUXDB_URL,
    token=INFLUXDB_TOKEN,
    org=INFLUXDB_ORG
)

# 告警级别定义
ALERT_LEVELS = {
    "CRITICAL": 4,
    "HIGH": 3,
    "MEDIUM": 2,
    "LOW": 1
}

def handle_alert(data):
    """处理告警通知"""
    alert_type = data.get("alert_type", "unknown")
    level = data.get("level", "LOW").upper()
    message = data.get("message", "No message")
    
    # 记录告警日志
    if level in ["CRITICAL", "HIGH"]:
        logger.critical(f"ALERT [{level}] {alert_type}: {message}")
    elif level == "MEDIUM":
        logger.warning(f"ALERT [{level}] {alert_type}: {message}")
    else:
        logger.info(f"ALERT [{level}] {alert_type}: {message}")
    
    # 预留Webhook接口（可扩展为钉钉/微信/邮件通知）
    webhook_url = os.environ.get("ALERT_WEBHOOK_URL")
    if webhook_url and level in ["CRITICAL", "HIGH"]:
        try:
            import requests
            payload = {
                "text": f"设备告警 [{level}]\n类型: {alert_type}\n消息: {message}"
            }
            requests.post(webhook_url, json=payload, timeout=5)
            logger.info(f"Alert webhook sent for {alert_type}")
        except Exception as e:
            logger.error(f"Failed to send alert webhook: {e}")

def verify_influxdb_connection():
    try:
        health = influx_client.health()
        if health.status == "pass":
            logger.info(f"InfluxDB connected successfully: {health.message}")
            return True
        else:
            logger.error(f"InfluxDB health check failed: {health.status} - {health.message}")
            return False
    except Exception as e:
        logger.error(f"Failed to connect to InfluxDB: {e}")
        return False

if not verify_influxdb_connection():
    logger.error("Cannot connect to InfluxDB, exiting...")
    exit(1)

# 批量写入配置
write_options = WriteOptions(
    batch_size=10,  # 每批最多10条记录
    flush_interval=5000,  # 每5秒刷新一次
    max_retries=3,  # 最大重试次数
    retry_interval=1000,  # 重试间隔1秒
    max_retry_delay=30000,  # 最大重试延迟30秒
    exponential_base=2  # 指数退避基数
)
write_api = influx_client.write_api(write_options=write_options)

def on_connect(client, userdata, flags, rc):
    if rc == 0:
        logger.info("Connected to MQTT broker")
        for topic in MQTT_TOPICS:
            client.subscribe(topic, qos=1)  # QoS 1: 至少一次送达
            logger.info(f"Subscribed to {topic} with QoS 1")
    else:
        logger.error(f"Failed to connect to MQTT broker with result code {rc}")

def on_disconnect(client, userdata, rc):
    if rc != 0:
        logger.warning(f"Unexpected MQTT disconnection (rc={rc}). Will auto-reconnect.")

def on_message(client, userdata, msg):
    try:
        data = json.loads(msg.payload.decode())
        method = data.get("method", "unknown")
        topic = msg.topic

        # 处理告警消息
        if topic == "device/alert":
            point = Point("device_alerts").tag("device", "imx6ull")
            if "alert_type" in data:
                point.field("alert_type", str(data["alert_type"]))
            if "message" in data:
                point.field("message", str(data["message"]))
            if "level" in data:
                point.field("level", str(data["level"]))
            try:
                write_api.write(bucket=INFLUXDB_BUCKET, record=point)
                logger.info(f"Written alert: {point}")
            except Exception as e:
                logger.error(f"Failed to write alert: {e}")
            handle_alert(data)
            return

        # 处理心跳消息
        if topic == "device/heartbeat":
            point = Point("device_heartbeat").tag("device", "imx6ull")
            if "cpu_usage" in data:
                point.field("cpu_usage", float(data["cpu_usage"]))
            if "mem_usage" in data:
                point.field("mem_usage", float(data["mem_usage"]))
            if "load_avg" in data:
                point.field("load_avg", float(data["load_avg"]))
            if "uptime" in data:
                point.field("uptime", int(data["uptime"]))
            try:
                write_api.write(bucket=INFLUXDB_BUCKET, record=point)
                logger.info(f"Written heartbeat: {point}")
            except Exception as e:
                logger.error(f"Failed to write heartbeat: {e}")
            return

        # 处理图片上传消息
        if method == "image_upload":
            event = data.get("event", "unknown")
            image_data = data.get("image_data", "")
            image_size = data.get("image_size", 0)
            timestamp = data.get("timestamp", int(time.time()))

            # 保存图片到本地
            if image_data:
                try:
                    import base64
                    img_bytes = base64.b64decode(image_data)
                    filename = f"images/{event}_{timestamp}.jpg"
                    os.makedirs("images", exist_ok=True)
                    with open(filename, "wb") as f:
                        f.write(img_bytes)
                    logger.info(f"Image saved: {filename} ({len(img_bytes)} bytes)")
                except Exception as e:
                    logger.error(f"Failed to save image: {e}")

            # 记录到InfluxDB
            point = Point("camera_images").tag("device", "imx6ull")
            point.field("event", event)
            point.field("image_size", image_size)
            point.field("timestamp", timestamp)
            try:
                write_api.write(bucket=INFLUXDB_BUCKET, record=point)
                logger.info(f"Written image event: {event}")
            except Exception as e:
                logger.error(f"Failed to write image event: {e}")

            # 触发告警通知（如果是烟雾报警）
            if event == "smoke_alert":
                handle_alert({
                    "alert_type": "smoke_with_image",
                    "level": "CRITICAL",
                    "message": f"Smoke alert with image captured ({image_size} bytes)"
                })
            return

        if method == "telemetry" and "data" in data:
            point = Point("sensor_data").tag("device", "imx6ull")
            d = data["data"]
            if "pir" in d:
                point.field("pir", int(d["pir"]))
            if "light" in d:
                point.field("light", int(d["light"]))
            if "humi" in d:
                point.field("humidity", int(d["humi"]))
            if "temp" in d:
                point.field("temperature", int(d["temp"]))
            if "relay" in d:
                point.field("relay", int(d["relay"]))
            if "relay2" in d:
                point.field("relay2", int(d["relay2"]))
            if "smoke_digital" in d:
                point.field("smoke_digital", int(d["smoke_digital"]))
            try:
                write_api.write(bucket=INFLUXDB_BUCKET, record=point)
                logger.info(f"Written telemetry: {point}")
            except Exception as e:
                logger.error(f"Failed to write telemetry: {e}")
            return

        if data.get("success") != 1:
            return

        point = Point("sensor_data").tag("device", "imx6ull")

        if method == "pir_read" and "data" in data and "pir" in data["data"]:
            point.field("pir", int(data["data"]["pir"]))
        elif method == "light_read" and "data" in data and "light" in data["data"]:
            light_val = data["data"]["light"]
            if isinstance(light_val, list):
                light_val = light_val[0] if light_val else 0
            point.field("light", int(light_val))
        elif method == "dht11_read" and "data" in data:
            humi = data["data"].get("humi")
            temp = data["data"].get("temp")
            if humi is not None:
                point.field("humidity", int(humi))
            if temp is not None:
                point.field("temperature", int(temp))
        elif method == "relay_control":
            point.field("relay_success", 1)
        elif method == "relay2_control":
            point.field("relay2_success", 1)
        elif method == "smoke_digital_read" and "data" in data and "smoke_digital" in data["data"]:
            point.field("smoke_digital", int(data["data"]["smoke_digital"]))
        else:
            logger.debug(f"Unhandled method: {method}")
            return

        try:
            write_api.write(bucket=INFLUXDB_BUCKET, record=point)
            logger.info(f"Written response: {point}")
        except Exception as e:
            logger.error(f"Failed to write response: {e}")

    except json.JSONDecodeError as e:
        logger.error(f"JSON parse error: {e}")
    except Exception as e:
        logger.exception(f"Error processing message: {e}")

mqtt_client = mqtt.Client()
mqtt_client.on_connect = on_connect
mqtt_client.on_disconnect = on_disconnect
mqtt_client.on_message = on_message
mqtt_client.username_pw_set(MQTT_USER, MQTT_PASS)
mqtt_client.reconnect_delay_set(min_delay=1, max_delay=60)

def cleanup():
    """清理资源"""
    try:
        write_api.close()
        influx_client.close()
        logger.info("InfluxDB connection closed")
    except Exception as e:
        logger.error(f"Error during cleanup: {e}")

import atexit
atexit.register(cleanup)

logger.info("Starting MQTT Loop...")
mqtt_client.connect(MQTT_BROKER, MQTT_PORT, 60)
mqtt_client.loop_forever()
